27 Type PopID(uint inBits, SubShapeID &outRemainder) const
28 {
29 Type mask_bits = Type((BiggerType(1) << inBits) - 1);
30 Type fill_bits = Type(BiggerType(cEmpty) << (MaxBits - inBits)); // Fill left side bits with 1 so that if there's no remainder all bits will be set, note that we do this using a BiggerType since on intel 0xffffffff << 32 == 0xffffffff
31 Type v = mValue & mask_bits;
32 outRemainder = SubShapeID(Type(BiggerType(mValue) >> inBits) | fill_bits);
33 return v;
34 }

66 void PushID(Type inValue, uint inFirstBit, uint inBits)
67 {
68 // First clear the bits
69 mValue &= ~(Type((BiggerType(1) << inBits) - 1) << inFirstBit);

71 // Then set them to the new value

26inline float CenterAngleAroundZero(float inV)
27{
28 if (inV < -JPH_PI)
29 {
30 do
31 inV += 2.0f * JPH_PI;
32 while (inV < -JPH_PI);
33 }
34 else if (inV > JPH_PI)
35 {
36 do
37 inV -= 2.0f * JPH_PI;
38 while (inV > JPH_PI);
39 }
40 JPH_ASSERT(inV >= -JPH_PI && inV <= JPH_PI);

42}

96inline uint CountTrailingZeros(uint32 inValue)
97{
98#if defined(JPH_CPU_X64)
99 #if defined(JPH_USE_TZCNT)
100 return _tzcnt_u32(inValue);
101 #elif defined(JPH_COMPILER_MSVC)
102 if (inValue == 0)
103 return 32;
104 unsigned long result;
105 _BitScanForward(&result, inValue);
106 return result;
107 #else
108 if (inValue == 0)
109 return 32;
110 return __builtin_ctz(inValue);
111 #endif
112#elif defined(JPH_CPU_ARM64)
113 return __builtin_clz(__builtin_bitreverse32(inValue));
114#else
115 #error Undefined
116#endif
117}

144inline uint CountBits(uint32 inValue)
145{
146#if defined(JPH_COMPILER_CLANG) || defined(JPH_COMPILER_GCC)
147 return __builtin_popcount(inValue);
148#elif defined(JPH_COMPILER_MSVC)
149 #if defined(JPH_USE_SSE4_2)
150 return _mm_popcnt_u32(inValue);
151 #else
152 inValue = inValue - ((inValue >> 1) & 0x55555555);
153 inValue = (inValue & 0x33333333) + ((inValue >> 2) & 0x33333333);
154 inValue = (inValue + (inValue >> 4)) & 0x0F0F0F0F;
155 return (inValue * 0x01010101) >> 24;
156 #endif
157#else
158 #error Undefined
159#endif
160}

163inline uint32 GetNextPowerOf2(uint32 inValue)
164{
165 return inValue <= 1? uint32(1) : uint32(1) << (32 - CountLeadingZeros(inValue - 1));

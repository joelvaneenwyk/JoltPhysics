175 inline bool SetBarrier(Barrier *inBarrier)
176 {
177 intptr_t barrier = 0;
178 if (mBarrier.compare_exchange_strong(barrier, reinterpret_cast<intptr_t>(inBarrier)))
179 return true;
180 JPH_ASSERT(barrier == cBarrierDoneState, "A job can only belong to 1 barrier");
181 return false;
182 }
183
185 inline uint32 Execute()
186 {
187 // Transition job to executing state
188 uint32 state = 0; // We can only start running with a dependency counter of 0
189 if (!mNumDependencies.compare_exchange_strong(state, cExecutingState))
190 return state; // state is updated by compare_exchange_strong to the current value
191
192 // Run the job function
193 {
194 JPH_PROFILE(mJobName, mColor.GetUInt32());
195 mJobFunction();
196 }
197
198 // Fetch the barrier pointer and exchange it for the done state, so we're sure that no barrier gets set after we want to call the callback
199 intptr_t barrier;
200 for (;;)
201 {
202 barrier = mBarrier;
203 if (mBarrier.compare_exchange_strong(barrier, cBarrierDoneState))
204 break;
205 }
206 JPH_ASSERT(barrier != cBarrierDoneState);
207
208 // Mark job as done
209 state = cExecutingState;
210 mNumDependencies.compare_exchange_strong(state, cDoneState);
211 JPH_ASSERT(state == cExecutingState);
212
213 // Notify the barrier after we've changed the job to the done state so that any thread reading the state after receiving the callback will see that the job has finished
214 if (barrier != 0)
215 reinterpret_cast<Barrier *>(barrier)->OnJobFinished(this);
216
217 return cDoneState;
218 }
219
221 inline bool CanBeExecuted() const { return mNumDependencies == 0; }
222
224 inline bool IsDone() const { return mNumDependencies == cDoneState; }

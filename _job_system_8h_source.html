173 inline bool SetBarrier(Barrier *inBarrier)
174 {
175 intptr_t barrier = 0;
176 if (mBarrier.compare_exchange_strong(barrier, reinterpret_cast<intptr_t>(inBarrier)))
177 return true;
178 JPH_ASSERT(barrier == cBarrierDoneState, "A job can only belong to 1 barrier");
179 return false;
180 }
181
183 inline uint32 Execute()
184 {
185 // Transition job to executing state
186 uint32 state = 0; // We can only start running with a dependency counter of 0
187 if (!mNumDependencies.compare_exchange_strong(state, cExecutingState))
188 return state; // state is updated by compare_exchange_strong to the current value
189
190 // Run the job function
191 {
192 JPH_PROFILE(mJobName, mColor.GetUInt32());
193 mJobFunction();
194 }
195
196 // Fetch the barrier pointer and exchange it for the done state, so we're sure that no barrier gets set after we want to call the callback
197 intptr_t barrier;
198 for (;;)
199 {
200 barrier = mBarrier;
201 if (mBarrier.compare_exchange_strong(barrier, cBarrierDoneState))
202 break;
203 }
204 JPH_ASSERT(barrier != cBarrierDoneState);
205
206 // Mark job as done
207 state = cExecutingState;
208 mNumDependencies.compare_exchange_strong(state, cDoneState);
209 JPH_ASSERT(state == cExecutingState);
210
211 // Notify the barrier after we've changed the job to the done state so that any thread reading the state after receiving the callback will see that the job has finished
212 if (barrier != 0)
213 reinterpret_cast<Barrier *>(barrier)->OnJobFinished(this);
214
215 return cDoneState;
216 }
217
219 inline bool CanBeExecuted() const { return mNumDependencies == 0; }

222 inline bool IsDone() const { return mNumDependencies == cDoneState; }

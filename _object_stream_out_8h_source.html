30 // Create the output stream

33 if (stream)
34 {
35 // Write the object to the stream

38 }

138template <class T>
139void OSWriteData(ObjectStreamOut &ioStream, const vector<T> &inArray)
140{
141 // Write size of array
142 ioStream.HintNextItem();
143 ioStream.WriteCount((uint32)inArray.size());
144
145 // Write data in array
146 ioStream.HintIndentUp();
147 for (const T &v : inArray)
148 OSWriteData(ioStream, v);
149 ioStream.HintIndentDown();

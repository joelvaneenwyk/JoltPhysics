44 virtual void * Allocate(uint inSize) override
45 {
46 if (inSize == 0)
47 {
48 return nullptr;
49 }
50 else
51 {
52 uint new_top = mTop + AlignUp(inSize, 16);
53 if (new_top > mSize)
54 JPH_CRASH; // Out of memory
55 void *address = mBase + mTop;
56 mTop = new_top;
57 return address;
58 }
59 }
60
61 // See: TempAllocator
62 virtual void Free(void *inAddress, uint inSize) override
63 {
64 if (inAddress == nullptr)
65 {
66 JPH_ASSERT(inSize == 0);
67 }
68 else
69 {
70 mTop -= AlignUp(inSize, 16);
71 if (mBase + mTop != inAddress)
72 JPH_CRASH; // Freeing in the wrong order
73 }
74 }
75
76private:
77 uint8 * mBase;
78 uint mSize;
79 uint mTop = 0;

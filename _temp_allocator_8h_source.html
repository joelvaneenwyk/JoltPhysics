28class TempAllocatorImpl final : public TempAllocator

34 mSize(inSize)

36 }

39 virtual ~TempAllocatorImpl() override
40 {
41 JPH_ASSERT(mTop == 0);
42 free(mBase);
43 }
44
45 // See: TempAllocator
46 virtual void * Allocate(uint inSize) override
47 {
48 if (inSize == 0)
49 {
50 return nullptr;
51 }
52 else
53 {
54 uint new_top = mTop + AlignUp(inSize, 16);
55 if (new_top > mSize)
56 JPH_CRASH; // Out of memory
57 void *address = mBase + mTop;
58 mTop = new_top;
59 return address;
60 }
61 }
62
63 // See: TempAllocator
64 virtual void Free(void *inAddress, uint inSize) override
65 {
66 if (inAddress == nullptr)
67 {
68 JPH_ASSERT(inSize == 0);
69 }
70 else
71 {
72 mTop -= AlignUp(inSize, 16);
73 if (mBase + mTop != inAddress)
74 JPH_CRASH; // Freeing in the wrong order
75 }
76 }
77
78private:
79 uint8 * mBase;
80 uint mSize;
81 uint mTop = 0;
82};

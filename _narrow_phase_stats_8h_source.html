41class TrackNarrowPhaseStat
42{
43public:
44 TrackNarrowPhaseStat(NarrowPhaseStat &inStat) :
45 mStat(inStat),
46 mParent(sRoot),
47 mStart(GetProcessorTickCount())
48 {
49 // Make this the new root of the chain
50 sRoot = this;
51 }
52
53 ~TrackNarrowPhaseStat()
54 {
55 uint64 delta_ticks = GetProcessorTickCount() - mStart;
56
57 // Notify parent of time spent in child
58 if (mParent != nullptr)
59 mParent->mStat.mChildTicks += delta_ticks;
60
61 // Increment stats at this level
62 mStat.mNumQueries++;
63 mStat.mTotalTicks += delta_ticks;
64
65 // Restore root pointer
66 JPH_ASSERT(sRoot == this);
67 sRoot = mParent;
68 }

70 NarrowPhaseStat & mStat;

72 uint64 mStart;

78class TrackNarrowPhaseCollector
79{
80public:

83 {
84 }
85
86 ~TrackNarrowPhaseCollector()
87 {
88 // Mark time spent in collector as 'child' time for the parent
89 uint64 delta_ticks = GetProcessorTickCount() - mStart;
90 if (TrackNarrowPhaseStat::sRoot != nullptr)
91 TrackNarrowPhaseStat::sRoot->mStat.mChildTicks += delta_ticks;
92
93 // Notify all parents of a hit
94 for (TrackNarrowPhaseStat *track = TrackNarrowPhaseStat::sRoot; track != nullptr; track = track->mParent)
95 track->mStat.mHitsReported++;

169template <class T>
170bool OSReadData(ObjectStreamIn &ioStream, vector<T> &inArray)
171{
172 bool continue_reading = true;
173
174 // Read array length
175 uint32 array_length;
176 continue_reading = ioStream.ReadCount(array_length);
177
178 // Read array items
179 if (continue_reading)
180 {
181 inArray.resize(array_length);
182 for (uint32 el = 0; el < array_length && continue_reading; ++el)
183 continue_reading = OSReadData(ioStream, inArray[el]);
184 }
185
186 return continue_reading;
187}
188
190template <class T, uint N>
191bool OSReadData(ObjectStreamIn &ioStream, StaticArray<T, N> &inArray)
192{
193 bool continue_reading = true;
194
195 // Read array length
196 uint32 array_length;
197 continue_reading = ioStream.ReadCount(array_length);
198
199 // Check if we can fit this many elements
200 if (array_length > N)
201 return false;
202
203 // Read array items
204 if (continue_reading)
205 {
206 inArray.resize(array_length);
207 for (uint32 el = 0; el < array_length && continue_reading; ++el)
208 continue_reading = OSReadData(ioStream, inArray[el]);
209 }
210
211 return continue_reading;
212}

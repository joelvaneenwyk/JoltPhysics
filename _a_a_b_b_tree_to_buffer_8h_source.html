49template <class TriangleCodec, class NodeCodec>
50class AABBTreeToBuffer
51{
52public:
54 using NodeHeader = typename NodeCodec::Header;
55
57 static const int HeaderSize = NodeCodec::HeaderSize;
58
60 static const int NumChildrenPerNode = NodeCodec::NumChildrenPerNode;
61
63 using TriangleHeader = typename TriangleCodec::TriangleHeader;
64
66 static const int TriangleHeaderSize = TriangleCodec::TriangleHeaderSize;
67
69 bool Convert(const VertexList &inVertices, const AABBTreeBuilder::Node *inRoot, AABBTreeToBufferStats &outStats, string &outError, EAABBTreeToBufferConvertMode inConvertMode = EAABBTreeToBufferConvertMode::DepthFirst)
70 {
71 const typename NodeCodec::EncodingContext node_ctx;
72 typename TriangleCodec::EncodingContext tri_ctx;
73
74 // Estimate the amount of memory required
75 uint tri_count = inRoot->GetTriangleCountInTree();
76 uint node_count = inRoot->GetNodeCount();
77 uint leaf_node_count = inRoot->GetLeafNodeCount();
78 uint nodes_size = node_ctx.GetPessimisticMemoryEstimate(node_count, leaf_node_count);
79 uint total_size = HeaderSize + TriangleHeaderSize + nodes_size + tri_ctx.GetPessimisticMemoryEstimate(tri_count);
80 mTree.reserve(total_size);
81
82 // Reset counters
83 mNodesSize = 0;
84
85 // Add headers
86 NodeHeader *header = HeaderSize > 0? mTree.Allocate<NodeHeader>() : nullptr;
87 TriangleHeader *triangle_header = TriangleHeaderSize > 0? mTree.Allocate<TriangleHeader>() : nullptr;
88
89 struct NodeData
90 {
91 NodeData() : mNode(nullptr), mNodeStart((uint)-1), mTriangleStart((uint)-1), mNumChildren(0), mParentChildNodeStart(nullptr), mParentTrianglesStart(nullptr) { }
92
93 const AABBTreeBuilder::Node * mNode; // Node that this entry belongs to
94 Vec3 mNodeBoundsMin; // Quantized node bounds
95 Vec3 mNodeBoundsMax;
96 uint mNodeStart; // Start of node in mTree
97 uint mTriangleStart; // Start of the triangle data in mTree
98 uint mNumChildren; // Number of children
99 uint mChildNodeStart[NumChildrenPerNode]; // Start of the children of the node in mTree
100 uint mChildTrianglesStart[NumChildrenPerNode]; // Start of the triangle data in mTree
101 uint * mParentChildNodeStart; // Where to store mNodeStart (to patch mChildNodeStart of my parent)
102 uint * mParentTrianglesStart; // Where to store mTriangleStart (to patch mChildTrianglesStart of my parent)
103 };
104
105 deque<NodeData *> to_process;
106 deque<NodeData *> to_process_triangles;
107 vector<NodeData> node_list;
108
109 node_list.reserve(node_count); // Needed to ensure that array is not reallocated, so we can keep pointers in the array
110
111 NodeData root;
112 root.mNode = inRoot;
113 root.mNodeBoundsMin = inRoot->mBounds.mMin;
114 root.mNodeBoundsMax = inRoot->mBounds.mMax;
115 node_list.push_back(root);
116 to_process.push_back(&node_list.back());
117
118 // Child nodes out of loop so we don't constantly realloc it
119 vector<const AABBTreeBuilder::Node *> child_nodes;
120 child_nodes.reserve(NumChildrenPerNode);
121
122 for (;;)
123 {
124 while (!to_process.empty())
125 {
126 // Get the next node to process
127 NodeData *node_data = nullptr;
128 switch (inConvertMode)
129 {
130 case EAABBTreeToBufferConvertMode::DepthFirst:
131 case EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast:
132 node_data = to_process.back();
133 to_process.pop_back();
134 break;
135
136 case EAABBTreeToBufferConvertMode::BreadthFirst:
137 case EAABBTreeToBufferConvertMode::BreadthFirstTrianglesLast:
138 node_data = to_process.front();
139 to_process.pop_front();
140 break;
141
142 default:
143 JPH_ASSERT(false);
144 break;
145 }
146
147 // Due to quantization box could have become bigger, not smaller
148 JPH_ASSERT(AABox(node_data->mNodeBoundsMin, node_data->mNodeBoundsMax).Contains(node_data->mNode->mBounds), "AABBTreeToBuffer: Bounding box became smaller!");
149
150 // Collect the first NumChildrenPerNode sub-nodes in the tree
151 child_nodes.clear(); // Won't free the memory
152 node_data->mNode->GetNChildren(NumChildrenPerNode, child_nodes);
153 node_data->mNumChildren = (uint)child_nodes.size();
154
155 // Fill in default child bounds
156 Vec3 child_bounds_min[NumChildrenPerNode], child_bounds_max[NumChildrenPerNode];
157 for (size_t i = 0; i < NumChildrenPerNode; ++i)
158 if (i < child_nodes.size())
159 {
160 child_bounds_min[i] = child_nodes[i]->mBounds.mMin;
161 child_bounds_max[i] = child_nodes[i]->mBounds.mMax;
162 }
163 else
164 {
165 child_bounds_min[i] = Vec3::sZero();
166 child_bounds_max[i] = Vec3::sZero();
167 }
168
169 // Start a new node
170 uint old_size = (uint)mTree.size();
171 node_data->mNodeStart = node_ctx.NodeAllocate(node_data->mNode, node_data->mNodeBoundsMin, node_data->mNodeBoundsMax, child_nodes, child_bounds_min, child_bounds_max, mTree, outError);
172 if (node_data->mNodeStart == uint(-1))
173 return false;
174 mNodesSize += (uint)mTree.size() - old_size;
175
176 if (node_data->mNode->HasChildren())
177 {
178 for (size_t i = 0; i < child_nodes.size(); ++i)
179 {
180 // Depth first: Insert in reverse order so we process left child first when taking nodes from the back
181 size_t idx = (inConvertMode == EAABBTreeToBufferConvertMode::DepthFirst || inConvertMode == EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast)?
182 child_nodes.size() - 1 - i : i;
183
184 // Due to quantization box could have become bigger, not smaller
185 JPH_ASSERT(AABox(child_bounds_min[idx], child_bounds_max[idx]).Contains(child_nodes[idx]->mBounds), "AABBTreeToBuffer: Bounding box became smaller!");
186
187 // Add child to list of nodes to be processed
188 NodeData child;
189 child.mNode = child_nodes[idx];
190 child.mNodeBoundsMin = child_bounds_min[idx];
191 child.mNodeBoundsMax = child_bounds_max[idx];
192 child.mParentChildNodeStart = &node_data->mChildNodeStart[idx];
193 child.mParentTrianglesStart = &node_data->mChildTrianglesStart[idx];
194 NodeData *old = &node_list[0];
195 node_list.push_back(child);
196 if (old != &node_list[0])
197 {
198 outError = "Internal Error: Array reallocated, memory corruption!";
199 return false;
200 }
201
202 switch (inConvertMode)
203 {
204 case EAABBTreeToBufferConvertMode::DepthFirst:
205 case EAABBTreeToBufferConvertMode::BreadthFirst:
206 to_process.push_back(&node_list.back());
207 break;
208
209 case EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast:
210 case EAABBTreeToBufferConvertMode::BreadthFirstTrianglesLast:
211 if (node_list.back().mNode->HasChildren())
212 to_process.push_back(&node_list.back());
213 else
214 to_process_triangles.push_back(&node_list.back());
215 break;
216 }
217 }
218 }
219 else
220 {
221 // Add triangles
222 node_data->mTriangleStart = tri_ctx.Pack(inVertices, node_data->mNode->mTriangles, node_data->mNodeBoundsMin, node_data->mNodeBoundsMax, mTree, outError);
223 if (node_data->mTriangleStart == uint(-1))
224 return false;
225 }
226
227 // Patch offset into parent
228 if (node_data->mParentChildNodeStart != nullptr)
229 {
230 *node_data->mParentChildNodeStart = node_data->mNodeStart;
231 *node_data->mParentTrianglesStart = node_data->mTriangleStart;
232 }
233 }
234
235 // If we've got triangles to process, loop again with just the triangles
236 if (to_process_triangles.empty())
237 break;
238 else
239 to_process.swap(to_process_triangles);
240 }
241
242 // Finalize all nodes
243 for (NodeData &n : node_list)
244 if (!node_ctx.NodeFinalize(n.mNode, n.mNodeStart, n.mTriangleStart, n.mNumChildren, n.mChildNodeStart, n.mChildTrianglesStart, mTree, outError))
245 return false;
246
247 // Finalize the triangles
248 tri_ctx.Finalize(triangle_header, mTree);
249
250 // Get stats
251 tri_ctx.GetStats(outStats.mTriangleCodecName, outStats.mVerticesPerTriangle);
252
253 // Validate that we reserved enough memory
254 if (nodes_size < mNodesSize)
255 {
256 outError = "Internal Error: Not enough memory reserved for nodes!";
257 return false;
258 }
259 if (total_size < (uint)mTree.size())
260 {
261 outError = "Internal Error: Not enough memory reserved for triangles!";
262 return false;
263 }
264
265 // Finalize the nodes
266 if (!node_ctx.Finalize(header, inRoot, node_list[0].mNodeStart, node_list[0].mTriangleStart, outError))
267 return false;
268
269 // Shrink the tree, this will invalidate the header and triangle_header variables
270 mTree.shrink_to_fit();
271
272 // Output stats
273 outStats.mTotalSize = (uint)mTree.size();
274 outStats.mNodesSize = mNodesSize;
275 outStats.mTrianglesSize = (uint)mTree.size() - mNodesSize;
276 outStats.mBytesPerTriangle = (float)mTree.size() / tri_count;
277
278 return true;
279 }

124 struct SubShape
125 {
130 bool FromSettings(const CompoundShapeSettings::SubShapeSettings &inSettings, ShapeResult &outResult)
131 {
132 if (inSettings.mShapePtr != nullptr)
133 {
134 // Use provided shape
135 mShape = inSettings.mShapePtr;
136 }
137 else
138 {
139 // Create child shape
140 ShapeResult child_result = inSettings.mShape->Create();
141 if (!child_result.IsValid())
142 {
143 outResult = child_result;
144 return false;
145 }
146 mShape = child_result.Get();
147 }
148
149 // Copy user data
150 mUserData = inSettings.mUserData;
151
152 SetTransform(inSettings.mPosition, inSettings.mRotation, Vec3::sZero() /* Center of mass not yet calculated */);
153 return true;
154 }
155
160 JPH_INLINE void SetTransform(Vec3Arg inPosition, QuatArg inRotation, Vec3Arg inCenterOfMass)
161 {
162 SetPositionCOM(inPosition - inCenterOfMass + inRotation * mShape->GetCenterOfMass());
163
164 mIsRotationIdentity = inRotation.IsClose(Quat::sIdentity()) || inRotation.IsClose(-Quat::sIdentity());
165 SetRotation(mIsRotationIdentity? Quat::sIdentity() : inRotation);
166 }
167
171 JPH_INLINE Mat44 GetLocalTransformNoScale(Vec3Arg inScale) const
172 {
173 JPH_ASSERT(IsValidScale(inScale));
174 return Mat44::sRotationTranslation(GetRotation(), inScale * GetPositionCOM());
175 }
176
178 inline bool IsValidScale(Vec3Arg inScale) const
179 {
180 // We can always handle uniform scale or identity rotations
181 if (mIsRotationIdentity || ScaleHelpers::IsUniformScale(inScale))
182 return true;
183
184 return ScaleHelpers::CanScaleBeRotated(GetRotation(), inScale);
185 }
186
188 inline Vec3 TransformScale(Vec3Arg inScale) const
189 {
190 // We don't need to transform uniform scale or if the rotation is identity
191 if (mIsRotationIdentity || ScaleHelpers::IsUniformScale(inScale))
192 return inScale;
193
194 return ScaleHelpers::RotateScale(GetRotation(), inScale);
195 }
196
198 JPH_INLINE void SetPositionCOM(Vec3Arg inPositionCOM)
199 {
200 inPositionCOM.StoreFloat3(&mPositionCOM);
201 }
202
204 JPH_INLINE Vec3 GetPositionCOM() const
205 {
206 return Vec3::sLoadFloat3Unsafe(mPositionCOM);
207 }
208
210 JPH_INLINE void SetRotation(QuatArg inRotation)
211 {
212 inRotation.StoreFloat3(&mRotation);
213 }
214
216 JPH_INLINE Quat GetRotation() const
217 {
218 return mIsRotationIdentity? Quat::sIdentity() : Quat::sLoadFloat3Unsafe(mRotation);
219 }
220
221 RefConst<Shape> mShape;
222 Float3 mPositionCOM;
223 Float3 mRotation;
224 uint32 mUserData;
225 bool mIsRotationIdentity;
226 // 3 padding bytes left
227 };

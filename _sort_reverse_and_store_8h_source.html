14JPH_INLINE int SortReverseAndStore(Vec4Arg inValues, float inMaxValue, UVec4 &ioIdentifiers, float *outValues)
15{
16 // Sort so that highest values are first (we want to first process closer hits and we process stack top to bottom)
17 Vec4::sSort4Reverse(inValues, ioIdentifiers);
18
19 // Count how many results are less than the max value
20 UVec4 closer = Vec4::sLess(inValues, Vec4::sReplicate(inMaxValue));
21 int num_results = closer.CountTrues();
22
23 // Shift the values so that only the ones that are less than max are kept
24 inValues = inValues.ReinterpretAsInt().ShiftComponents4Minus(num_results).ReinterpretAsFloat();
25 ioIdentifiers = ioIdentifiers.ShiftComponents4Minus(num_results);
26
27 // Store the values
28 inValues.StoreFloat4((Float4 *)outValues);
29
30 return num_results;
31}

109 inline SubShapeID MakeSubShapeIDRelativeToShape(const SubShapeID &inSubShapeID) const
110 {
111 // Take off the sub shape ID part that comes from mSubShapeIDCreator and validate that it is the same
112 SubShapeID sub_shape_id;
113 uint num_bits_written = mSubShapeIDCreator.GetNumBitsWritten();
114 JPH_IF_ENABLE_ASSERTS(uint32 root_id =) inSubShapeID.PopID(num_bits_written, sub_shape_id);
115 JPH_ASSERT(root_id == (mSubShapeIDCreator.GetID().GetValue() & ((1 << num_bits_written) - 1)));
116 return sub_shape_id;
117 }

121 inline Vec3 GetWorldSpaceSurfaceNormal(const SubShapeID &inSubShapeID, Vec3Arg inPosition) const
122 {
123 Mat44 inv_com = GetInverseCenterOfMassTransform();
124 Vec3 scale = GetShapeScale(); // See comment at ScaledShape::GetSurfaceNormal for the math behind the scaling of the normal
125 return inv_com.Multiply3x3Transposed(mShape->GetSurfaceNormal(MakeSubShapeIDRelativeToShape(inSubShapeID), (inv_com * inPosition) / scale) / scale).Normalized();
126 }

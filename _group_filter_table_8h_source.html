43 int GetBit(SubGroupID inSubGroup1, SubGroupID inSubGroup2) const
44 {
45 JPH_ASSERT(inSubGroup1 != inSubGroup2);
46
47 // We store the lower left half only, so swap the inputs when trying to access the top right half
48 if (inSubGroup1 > inSubGroup2)
49 swap(inSubGroup1, inSubGroup2);
50
51 JPH_ASSERT(inSubGroup2 < mNumSubGroups);
52
53 // Calculate at which bit the entry for this pair resides
54 // We use the fact that a row always starts at inSubGroup2 * (inSubGroup2 - 1) / 2
55 // (this is the amount of bits needed to store a table of inSubGroup2 entries)
56 return (inSubGroup2 * (inSubGroup2 - 1)) / 2 + inSubGroup1;
57 }

59public:
61 explicit GroupFilterTable(uint inNumSubGroups = 0) :
62 mNumSubGroups(inNumSubGroups)
63 {
64 // By default everything collides
65 int table_size = ((inNumSubGroups * (inNumSubGroups - 1)) / 2 + 7) / 8;
66 mTable.resize(table_size, 0xff);
67 }

73 void DisableCollision(SubGroupID inSubGroup1, SubGroupID inSubGroup2)
74 {
75 int bit = GetBit(inSubGroup1, inSubGroup2);
76 mTable[bit >> 3] &= (0xff ^ (1 << (bit & 0b111)));
77 }

80 void EnableCollision(SubGroupID inSubGroup1, SubGroupID inSubGroup2)
81 {
82 int bit = GetBit(inSubGroup1, inSubGroup2);
83 mTable[bit >> 3] |= 1 << (bit & 0b111);

87 inline bool IsCollisionEnabled(SubGroupID inSubGroup1, SubGroupID inSubGroup2) const
88 {
89 // Test if the bit is set for this group pair
90 int bit = GetBit(inSubGroup1, inSubGroup2);
91 return (mTable[bit >> 3] & (1 << (bit & 0b111))) != 0;
92 }

95 virtual bool CanCollide(const CollisionGroup &inGroup1, const CollisionGroup &inGroup2) const override
96 {
97 // If one of the groups is cInvalidGroup the objects will collide (note that the if following this if will ensure that group2 is not cInvalidGroup)
98 if (inGroup1.GetGroupID() == CollisionGroup::cInvalidGroup)
99 return true;
100
101 // If the objects are in different groups, they collide
102 if (inGroup1.GetGroupID() != inGroup2.GetGroupID())
103 return true;
104
105 // If the collision filters do not match, but they're in the same group we ignore the collision
106 if (inGroup1.GetGroupFilter() != inGroup2.GetGroupFilter())
107 return false;
108
109 // If they are in the same sub group, they don't collide
110 if (inGroup1.GetSubGroupID() == inGroup2.GetSubGroupID())
111 return false;
112
113 // Check the bit table
114 return IsCollisionEnabled(inGroup1.GetSubGroupID(), inGroup2.GetSubGroupID());
115 }

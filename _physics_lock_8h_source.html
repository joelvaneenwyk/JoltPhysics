35 static inline void sCheckUnlock(EPhysicsLockTypes inType)
36 {
37 JPH_ASSERT((sLockedMutexes & (uint32)inType) != 0, "Mutex was not locked!");
38 sLockedMutexes = sLockedMutexes & ~(uint32)inType;

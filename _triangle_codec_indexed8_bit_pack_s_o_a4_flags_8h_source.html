112 uint Pack(const VertexList &inVertices, const IndexedTriangleList &inTriangles, Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, ByteBuffer &ioBuffer, string &outError)
113 {
114 // Determine position of triangles start
115 uint offset = (uint)ioBuffer.size();
116
117 // Update stats
118 uint tri_count = (uint)inTriangles.size();
119 mNumTriangles += tri_count;
120
121 // Allocate triangle block header
122 TriangleBlockHeader *header = ioBuffer.Allocate<TriangleBlockHeader>();
123
124 // Compute first vertex that this batch will use (ensuring there's enough room if none of the vertices are shared)
125 uint start_vertex = Clamp((int)mVertices.size() - 256 + (int)tri_count * 3, 0, (int)mVertices.size());
126
127 // Store the start vertex offset, this will later be patched to give the delta offset relative to the triangle block
128 mOffsetsToPatch.push_back(uint((uint8 *)&header->mOffsetToVertices - &ioBuffer[0]));
129 header->mOffsetToVertices = start_vertex * sizeof(VertexData);
130
131 // Pack vertices
132 uint padded_triangle_count = AlignUp(tri_count, 4);
133 for (uint t = 0; t < padded_triangle_count; t += 4)
134 {
135 TriangleBlock *block = ioBuffer.Allocate<TriangleBlock>();
136 for (uint vertex_nr = 0; vertex_nr < 3; ++vertex_nr)
137 for (uint block_tri_idx = 0; block_tri_idx < 4; ++block_tri_idx)
138 {
139 // Fetch vertex index. Create degenerate triangles for padding triangles.
140 bool triangle_available = t + block_tri_idx < tri_count;
141 uint32 src_vertex_index = triangle_available? inTriangles[t + block_tri_idx].mIdx[vertex_nr] : inTriangles[tri_count - 1].mIdx[0];
142
143 // Check if we've seen this vertex before and if it is in the range that we can encode
144 uint32 vertex_index;
145 VertexMap::const_iterator found = mVertexMap.find(src_vertex_index);
146 if (found == mVertexMap.end() || found->second < start_vertex)
147 {
148 // Add vertex
149 vertex_index = (uint32)mVertices.size();
150 mVertexMap[src_vertex_index] = vertex_index;
151 mVertices.push_back(inVertices[src_vertex_index]);
152 }
153 else
154 {
155 // Reuse vertex
156 vertex_index = found->second;
157 }
158
159 // Store vertex index
160 uint32 vertex_offset = vertex_index - start_vertex;
161 if (vertex_offset > 0xff)
162 {
163 outError = "TriangleCodecIndexed8BitPackSOA4Flags: Offset doesn't fit in 8 bit";
164 return uint(-1);
165 }
166 block->mIndices[vertex_nr][block_tri_idx] = (uint8)vertex_offset;
167
168 // Store flags
169 uint32 flags = triangle_available? inTriangles[t + block_tri_idx].mMaterialIndex : 0;
170 if (flags > 0xff)
171 {
172 outError = "TriangleCodecIndexed8BitPackSOA4Flags: Material index doesn't fit in 8 bit";
173 return uint(-1);
174 }
175 block->mFlags[block_tri_idx] = (uint8)flags;
176 }
177 }
178
179 return offset;
180 }

183 void Finalize(TriangleHeader *ioHeader, ByteBuffer &ioBuffer) const
184 {
185 // Check if anything to do
186 if (mVertices.empty())
187 return;
188
189 // Align buffer to 4 bytes
190 uint vertices_idx = (uint)ioBuffer.Align(4);
191
192 // Patch the offsets
193 for (uint o : mOffsetsToPatch)
194 *ioBuffer.Get<uint32>(o) += vertices_idx - o;
195
196 // Calculate bounding box
197 AABox bounds;
198 for (const Float3 &v : mVertices)
199 bounds.Encapsulate(Vec3(v));
200
201 // Compress vertices
202 VertexData *vertices = ioBuffer.Allocate<VertexData>(mVertices.size());
203 Vec3 compress_scale = Vec3::sReplicate(COMPONENT_MASK) / Vec3::sMax(bounds.GetSize(), Vec3::sReplicate(1.0e-20f));
204 for (const Float3 &v : mVertices)
205 {
206 UVec4 c = ((Vec3(v) - bounds.mMin) * compress_scale + Vec3::sReplicate(0.5f)).ToInt();
207 JPH_ASSERT(c.GetX() <= COMPONENT_MASK);
208 JPH_ASSERT(c.GetY() <= COMPONENT_MASK);
209 JPH_ASSERT(c.GetZ() <= COMPONENT_MASK);
210 vertices->mVertexXY = c.GetX() + (c.GetY() << COMPONENT_Y1);
211 vertices->mVertexZY = c.GetZ() + ((c.GetY() >> COMPONENT_Y1_BITS) << COMPONENT_Y2);
212 ++vertices;
213 }
214
215 // Store decompression information
216 bounds.mMin.StoreFloat3(&ioHeader->mOffset);
217 (bounds.GetSize() / Vec3::sReplicate(COMPONENT_MASK)).StoreFloat3(&ioHeader->mScale);
218 }

241 JPH_INLINE void Unpack(const VertexData *inVertices, UVec4 inIndex, Vec4 &outX, Vec4 &outY, Vec4 &outZ) const
242 {
243 // Get compressed data
244 UVec4 c1 = UVec4::sGatherInt4<8>(&inVertices->mVertexXY, inIndex);
245 UVec4 c2 = UVec4::sGatherInt4<8>(&inVertices->mVertexZY, inIndex);
246
247 // Unpack the x y and z component
248 UVec4 xc = UVec4::sAnd(c1, UVec4::sReplicate(COMPONENT_MASK));
249 UVec4 yc = UVec4::sOr(c1.LogicalShiftRight<COMPONENT_Y1>(), c2.LogicalShiftRight<COMPONENT_Y2>().LogicalShiftLeft<COMPONENT_Y1_BITS>());
250 UVec4 zc = UVec4::sAnd(c2, UVec4::sReplicate(COMPONENT_MASK));
251
252 // Convert to float
253 outX = Vec4::sFusedMultiplyAdd(xc.ToFloat(), mScaleX, mOffsetX);
254 outY = Vec4::sFusedMultiplyAdd(yc.ToFloat(), mScaleY, mOffsetY);
255 outZ = Vec4::sFusedMultiplyAdd(zc.ToFloat(), mScaleZ, mOffsetZ);
256 }

270 JPH_INLINE void Unpack(Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, const void *inTriangleStart, uint32 inNumTriangles, Vec3 *outTriangles) const
271 {
272 JPH_ASSERT(inNumTriangles > 0);
273 const TriangleBlockHeader *header = reinterpret_cast<const TriangleBlockHeader *>(inTriangleStart);
274 const VertexData *vertices = header->GetVertexData();
275 const TriangleBlock *t = header->GetTriangleBlock();
276 const TriangleBlock *end = t + ((inNumTriangles + 3) >> 2);
277
278 int triangles_left = inNumTriangles;
279
280 do
281 {
282 // Get the indices for the three vertices (reads 4 bytes extra, but these are the flags so that's ok)
283 UVec4 indices = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&t->mIndices[0]));
284 UVec4 iv1 = indices.Expand4Byte0();
285 UVec4 iv2 = indices.Expand4Byte4();
286 UVec4 iv3 = indices.Expand4Byte8();
287
288 // Decompress the triangle data
289 Vec4 v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z;
290 Unpack(vertices, iv1, v1x, v1y, v1z);
291 Unpack(vertices, iv2, v2x, v2y, v2z);
292 Unpack(vertices, iv3, v3x, v3y, v3z);
293
294 // Transpose it so we get normal vectors
295 Mat44 v1 = Mat44(v1x, v1y, v1z, Vec4::sZero()).Transposed();
296 Mat44 v2 = Mat44(v2x, v2y, v2z, Vec4::sZero()).Transposed();
297 Mat44 v3 = Mat44(v3x, v3y, v3z, Vec4::sZero()).Transposed();
298
299 // Store triangle data
300 for (int i = 0; i < 4 && triangles_left > 0; ++i, --triangles_left)
301 {
302 *outTriangles++ = v1.GetColumn3(i);
303 *outTriangles++ = v2.GetColumn3(i);
304 *outTriangles++ = v3.GetColumn3(i);
305 }
306
307 ++t;
308 }
309 while (t < end);
310 }

313 JPH_INLINE float TestRay(Vec3Arg inRayOrigin, Vec3Arg inRayDirection, Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, const void *inTriangleStart, uint32 inNumTriangles, float inClosest, uint32 &outClosestTriangleIndex) const
314 {
315 JPH_ASSERT(inNumTriangles > 0);
316 const TriangleBlockHeader *header = reinterpret_cast<const TriangleBlockHeader *>(inTriangleStart);
317 const VertexData *vertices = header->GetVertexData();
318 const TriangleBlock *t = header->GetTriangleBlock();
319 const TriangleBlock *end = t + ((inNumTriangles + 3) >> 2);
320
321 Vec4 closest = Vec4::sReplicate(inClosest);
322 UVec4 closest_triangle_idx = UVec4::sZero();
323
324 UVec4 start_triangle_idx = UVec4::sZero();
325 do
326 {
327 // Get the indices for the three vertices (reads 4 bytes extra, but these are the flags so that's ok)
328 UVec4 indices = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&t->mIndices[0]));
329 UVec4 iv1 = indices.Expand4Byte0();
330 UVec4 iv2 = indices.Expand4Byte4();
331 UVec4 iv3 = indices.Expand4Byte8();
332
333 // Decompress the triangle data
334 Vec4 v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z;
335 Unpack(vertices, iv1, v1x, v1y, v1z);
336 Unpack(vertices, iv2, v2x, v2y, v2z);
337 Unpack(vertices, iv3, v3x, v3y, v3z);
338
339 // Perform ray vs triangle test
340 Vec4 distance = RayTriangle4(inRayOrigin, inRayDirection, v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z);
341
342 // Update closest with the smaller values
343 UVec4 smaller = Vec4::sLess(distance, closest);
344 closest = Vec4::sSelect(closest, distance, smaller);
345
346 // Update triangle index with the smallest values
347 UVec4 triangle_idx = start_triangle_idx + UVec4(0, 1, 2, 3);
348 closest_triangle_idx = UVec4::sSelect(closest_triangle_idx, triangle_idx, smaller);
349
350 // Next block
351 ++t;
352 start_triangle_idx += UVec4::sReplicate(4);
353 }
354 while (t < end);
355
356 // Get the smallest component
357 Vec4::sSort4(closest, closest_triangle_idx);
358 outClosestTriangleIndex = closest_triangle_idx.GetX();
359 return closest.GetX();
360 }

363 inline void GetTriangle(const void *inTriangleStart, uint32 inTriangleIdx, Vec3 &outV1, Vec3 &outV2, Vec3 &outV3) const
364 {
365 const TriangleBlockHeader *header = reinterpret_cast<const TriangleBlockHeader *>(inTriangleStart);
366 const VertexData *vertices = header->GetVertexData();
367 const TriangleBlock *block = header->GetTriangleBlock() + (inTriangleIdx >> 2);
368 uint32 block_triangle_idx = inTriangleIdx & 0b11;
369
370 // Get the 3 vertices
371 const VertexData &v1 = vertices[block->mIndices[0][block_triangle_idx]];
372 const VertexData &v2 = vertices[block->mIndices[1][block_triangle_idx]];
373 const VertexData &v3 = vertices[block->mIndices[2][block_triangle_idx]];
374
375 // Pack the vertices
376 UVec4 c1(v1.mVertexXY, v2.mVertexXY, v3.mVertexXY, 0);
377 UVec4 c2(v1.mVertexZY, v2.mVertexZY, v3.mVertexZY, 0);
378
379 // Unpack the x y and z component
380 UVec4 xc = UVec4::sAnd(c1, UVec4::sReplicate(COMPONENT_MASK));
381 UVec4 yc = UVec4::sOr(c1.LogicalShiftRight<COMPONENT_Y1>(), c2.LogicalShiftRight<COMPONENT_Y2>().LogicalShiftLeft<COMPONENT_Y1_BITS>());
382 UVec4 zc = UVec4::sAnd(c2, UVec4::sReplicate(COMPONENT_MASK));
383
384 // Convert to float
385 Vec4 vx = Vec4::sFusedMultiplyAdd(xc.ToFloat(), mScaleX, mOffsetX);
386 Vec4 vy = Vec4::sFusedMultiplyAdd(yc.ToFloat(), mScaleY, mOffsetY);
387 Vec4 vz = Vec4::sFusedMultiplyAdd(zc.ToFloat(), mScaleZ, mOffsetZ);
388
389 // Transpose it so we get normal vectors
390 Mat44 trans = Mat44(vx, vy, vz, Vec4::sZero()).Transposed();
391 outV1 = trans.GetAxisX();
392 outV2 = trans.GetAxisY();
393 outV3 = trans.GetAxisZ();
394 }
